' 全局变量
DimP g_hHook As HWINEVENTHOOK
DimP g_hWndLastForeground As HWND
Const WM_COPY = &H301
Const WM_PASTE = &H302

' WinEvent 回调函数
Sub MyWinEventProc stdcall (ByVal hHook As HWINEVENTHOOK, ByVal Event As DWORD, ByVal hwnd As hwnd, ByVal idObject As Long, ByVal idChild As Long, ByVal dwEventThread As DWORD, ByVal dwmsEventTime As DWORD )
    If Event = &H3 Then   ' EVENT_SYSTEM_FOREGROUND
       g_hWndLastForeground = GetForegroundWindow()
       Print "MyWinEventProc: event=" & Event & ", hwnd=" & hwnd & ", saved foreground=" & g_hWndLastForeground
    End If
End Sub


' 根据索引获取左右符号（索引从1开始，对应你的符号对数组）
Function GetSymbolPair(idx As Integer, ByRef leftSym As String, ByRef rightSym As String) As Boolean
    Select Case idx
        Case 1  ' 中文双引号
            leftSym = "“"
            rightSym = "”"
        Case 2  ' 英文双引号
            leftSym = """"
            rightSym = """"
        Case 3  ' 中文单引号
            leftSym = "‘"
            rightSym = "’"
        Case 4  ' 英文单引号
            leftSym = "'"
            rightSym = "'"
        Case 5  ' 中文书名号
            leftSym = "《"
            rightSym = "》"
        Case 6  ' 英文尖括号
            leftSym = "<"
            rightSym = ">"
        Case 7  ' 中文方括号
            leftSym = "【"
            rightSym = "】"
        Case 8  ' 英文方括号
            leftSym = "["
            rightSym = "]"
        Case 9  ' 中文花括号「」
            leftSym = "「"
            rightSym = "」"
        Case 10 ' 英文花括号
            leftSym = "{"
            rightSym = "}"
        Case 11 ' 中文圆括号（）
            leftSym = "（"
            rightSym = "）"
        Case 12 ' 英文圆括号
            leftSym = "("
            rightSym = ")"
        Case Else
            Return False
    End Select
    Return True
End Function

Sub InsertSymbolPair(idx As Integer)
   Print "=== InsertSymbolPair start, idx=" & idx
   Print "g_hWndLastForeground=" & g_hWndLastForeground
   Dim leftSym As String, rightSym As String
   If Not GetSymbolPair(idx, leftSym, rightSym) Then Exit Sub
   
   If g_hWndLastForeground = 0 Then
      MsgBox "未能获取到原窗口，请确保已从其他程序切换到本程序。"
      Exit Sub
   End If
   
   'SetForegroundWindow(g_hWndLastForeground)
   Dim ret As Long
   ret = SetForegroundWindow(g_hWndLastForeground)
   Print "SetForegroundWindow ret="      & ret
   Print "Current foreground after Set=" & GetForegroundWindow()
   
   Sleep(100)
   
   KeyPressOR(17, 67) ' Ctrl+C
   Sleep(50)
   
   Dim selectedText As String = FF_ClipboardGetText()
   Print "selectedText='" & selectedText & "'"
   Dim newText      As String
   
   If selectedText <> "" Then
      newText = leftSym & selectedText & rightSym
   Else
      newText = leftSym & rightSym
   End If
   
   FF_ClipboardSetText(newText)
   Sleep(50)
   KeyPressOR(17, 86) ' Ctrl+V
   Sleep(100)
   Print "Paste done."
   
   ' 无选中文本时，将光标移到引号中间
   If selectedText = "" Then
      Sleep(50)
      KeyPressOR(0, 37) ' 左箭头（需KeyPressOR支持第一个参数0，否则改用keybd_event）
   End If
End Sub
