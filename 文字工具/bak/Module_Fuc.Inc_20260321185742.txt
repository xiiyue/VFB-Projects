' 全局变量
DimP g_hHook As HWINEVENTHOOK
DimP g_hWndCurrent As HWND      ' 当前前台窗口
DimP g_hWndLast As HWND          ' 上一次的前台窗口（即我们需要的原窗口）

' WinEvent 回调函数
Sub MyWinEventProc stdcall(ByVal hHook As HWINEVENTHOOK, ByVal Event As DWORD, ByVal hwnd As hwnd, ByVal idObject As Long, ByVal idChild As Long, ByVal dwEventThread As DWORD, ByVal dwmsEventTime As DWORD)
   If Event = &H3 Then ' EVENT_SYSTEM_FOREGROUND
      ' 更新当前窗口为新窗口
      g_hWndLast = g_hWndCurrent
      g_hWndCurrent = hwnd
   End If
End Sub

' ============================================
' 获取选中的文本，return 选中的文本
' ============================================
Function GetSelectedText() As String
    If g_hWndLast = 0 Then
        MsgBox "未能获取到原窗口，请确保已从其他程序切换到本程序。"
        Return ""
    End If

    ' 激活原窗口
    SetForegroundWindow(g_hWndLast)
    Sleep(100)

    ' 附加线程输入，获取焦点控件句柄
    Dim tid As Long = GetWindowThreadProcessId(g_hWndLast, 0)
    AttachThreadInput(GetCurrentThreadId(), tid, True)
    Dim hFocus As HWND = GetFocus()
    AttachThreadInput(GetCurrentThreadId(), tid, False)
    
    If hFocus = 0 Then
        MsgBox "无法获取目标编辑框句柄。"
        Return ""
    End If

    ' 模拟 Ctrl+C 复制
    keybd_event(VK_CONTROL, 0, 0, 0)
    keybd_event(Asc("C"), 0, 0, 0)
    keybd_event(Asc("C"), 0, KEYEVENTF_KEYUP, 0)
    keybd_event(VK_CONTROL, 0, KEYEVENTF_KEYUP, 0)
    Sleep(100)

    ' 读取剪贴板
    Dim selectedText As String = FF_ClipboardGetText()
    If selectedText = "" Then
        MsgBox "复制失败，请检查目标窗口是否支持复制。"
        Return ""
    End If

    Return selectedText
End Function

' ============================================
' 处理结果发送回去，参数为新文本，没有return
' ============================================
Sub ReplaceSelectedText(newText As String)
    ' 将新文本写入剪贴板
    AfxSetClipboardText(newText)
    Sleep(50)

    ' 模拟 Ctrl+V 粘贴
    keybd_event(VK_CONTROL, 0, 0, 0)
    keybd_event(Asc("V"), 0, 0, 0)
    keybd_event(Asc("V"), 0, KEYEVENTF_KEYUP, 0)
    keybd_event(VK_CONTROL, 0, KEYEVENTF_KEYUP, 0)
    Sleep(100)
End Sub

' ============================================
' 获取选中文本GetSelectedText → 处理(原文本) → 发送ReplaceSelectedText
' ============================================
Sub ProcessSelectedText(processFunc As Function(ByVal txt As String) As String)
    Dim originalText As String = GetSelectedText()
    If originalText = "" Then Exit Sub

    Dim newText As String = processFunc(originalText)
    If newText = originalText Then
        ' 文本无变化，可跳过粘贴
        Exit Sub
    End If
    ReplaceSelectedText(newText)
End Sub
' ============================================
' 获取选中文本GetSelectedText → 处理(原文本,参数) → 发送ReplaceSelectedText
' ============================================
Type ProcessFuncWithInt As Function(ByVal txt As String, ByVal i As Integer) As String
Sub ProcessSelectedTextWithInt(processFunc As ProcessFuncWithInt, i As Integer)
    Dim originalText As String = GetSelectedText()
    If originalText = "" Then Exit Sub

    Dim newText As String = processFunc(originalText, i)
    If newText = originalText Then
        Exit Sub
    End If
    ReplaceSelectedText(newText)
End Sub


' 符号对的辅助函数
Function GetSymbolPair(idx As Integer, ByRef leftSym As String, ByRef rightSym As String) As Boolean
    Select Case idx
        Case 1  ' 中文双引号
            leftSym = "“"
            rightSym = "”"
        Case 2  ' 英文双引号
            leftSym = """"
            rightSym = """"
        Case 3  ' 中文单引号
            leftSym = "‘"
            rightSym = "’"
        Case 4  ' 英文单引号
            leftSym = "'"
            rightSym = "'"
        Case 5  ' 中文书名号
            leftSym = "《"
            rightSym = "》"
        Case 6  ' 英文尖括号
            leftSym = "<"
            rightSym = ">"
        Case 7  ' 中文方括号
            leftSym = "【"
            rightSym = "】"
        Case 8  ' 英文方括号
            leftSym = "["
            rightSym = "]"
        Case 9  ' 中文花括号「」
            leftSym = "「"
            rightSym = "」"
        Case 10 ' 英文花括号
            leftSym = "{"
            rightSym = "}"
        Case 11 ' 中文圆括号（）
            leftSym = "（"
            rightSym = "）"
        Case 12 ' 英文圆括号
            leftSym = "("
            rightSym = ")"
        Case Else
            Return False
    End Select
    Return True
End Function

' 符号对的实现
Function InsertSymbolPair(oldtext As String, idx As Integer) As String
   Dim leftSym As String, rightSym As String
   If Not GetSymbolPair(idx, leftSym, rightSym) Then
      Return oldtext ' 获取失败时，返回原文本，不做修改
   End If
   Return leftSym & oldtext & rightSym
End Function



' 函数功能：自动检测并转换中文简繁
' 参数：oldtext - 待转换的字符串
' 返回值：转换后的字符串（若原为简体则返回繁体，若原为繁体则返回简体，无法判断时返回原文本）
Function ConvertChinese(oldtext As String) As String
    Dim As Integer srcLen = Len(oldtext)
    If srcLen = 0 Then Return ""

    ' 1. 分别获取转换为简体、繁体后的目标长度
    Dim As Integer simpLen = LCMapString(LOCALE_SYSTEM_DEFAULT, LCMAP_SIMPLIFIED_CHINESE, _
        StrPtr(oldtext), srcLen, NULL, 0)
    Dim As Integer tradLen = LCMapString(LOCALE_SYSTEM_DEFAULT, LCMAP_TRADITIONAL_CHINESE, _
        StrPtr(oldtext), srcLen, NULL, 0)

    If simpLen = 0 Or tradLen = 0 Then Return ""   ' 转换失败

    ' 2. 分配缓冲区并执行转换
    Dim As String simpText = Space(simpLen)
    Dim As String tradText = Space(tradLen)

    LCMapString(LOCALE_SYSTEM_DEFAULT, LCMAP_SIMPLIFIED_CHINESE, _
        StrPtr(oldtext), srcLen, StrPtr(simpText), simpLen)
    LCMapString(LOCALE_SYSTEM_DEFAULT, LCMAP_TRADITIONAL_CHINESE, _
        StrPtr(oldtext), srcLen, StrPtr(tradText), tradLen)

    ' 3. 统计原文本与两个转换结果之间相同字符的数量
    Dim As Integer sameWithSimp = 0, sameWithTrad = 0
    For i As Integer = 1 To srcLen
        If Mid(oldtext, i, 1) = Mid(simpText, i, 1) Then sameWithSimp += 1
        If Mid(oldtext, i, 1) = Mid(tradText, i, 1) Then sameWithTrad += 1
    Next

    ' 4. 根据相似度判断方向，返回另一种形式
    If sameWithSimp >= sameWithTrad Then
        Return tradText    ' 原文本更接近简体 → 转为繁体
    Else
        Return simpText    ' 原文本更接近繁体 → 转为简体
    End If
End Function
