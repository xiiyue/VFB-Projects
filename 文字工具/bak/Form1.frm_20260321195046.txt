'[Form1]事件 : 窗口完全显示后。
Sub Form1_Shown(hWndForm As hWnd, UserData As Integer)
   ' 初始化当前前台窗口
    g_hWndCurrent = GetForegroundWindow()
    ' 安装钩子
    g_hHook = SetWinEventHook(&H3, &H3, 0, @MyWinEventProc, 0, 0, 0)
    If g_hHook = 0 Then
        MsgBox "无法监听窗口切换，功能可能受限。"
     End If   
   ' 初始化
   Init_符号对()
End Sub

'[Form1]事件 : 即将销毁窗口
Sub Form1_WM_Destroy(hWndForm As hWnd)
    If g_hHook <> 0 Then
        UnhookWinEvent(g_hHook)
        g_hHook = 0
    End If
End Sub

'[Form1.Combo_符号对]事件 : 列表框中更改当前选择时
Sub Form1_Combo_符号对_CBN_SelChange(hWndForm As hWnd, hWndControl As hWnd)
   Dim idx As Integer = Combo_符号对.ListIndex
   If idx >= 1 Then
     ProcessSelectedTextWithInt(@InsertSymbolPair , idx) 
   End If
End Sub

'hWndControl 当前控件的句柄(也是窗口句柄，如果多开本窗口，必须 Me.控件名.hWndForm = hWndForm 后才可以执行后续操作本控件的代码 )

   ProcessSelectedText(@ConvertChinese)
End Sub
