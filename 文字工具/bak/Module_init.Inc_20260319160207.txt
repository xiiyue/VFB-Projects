' 根据索引获取左右符号（索引从1开始，对应你的符号对数组）
Function GetSymbolPair(idx As Integer, ByRef leftSym As String, ByRef rightSym As String) As Boolean
    Select Case idx
        Case 1  ' 中文双引号
            leftSym = "“"
            rightSym = "”"
        Case 2  ' 英文双引号
            leftSym = """"
            rightSym = """"
        Case 3  ' 中文单引号
            leftSym = "‘"
            rightSym = "’"
        Case 4  ' 英文单引号
            leftSym = "'"
            rightSym = "'"
        Case 5  ' 中文书名号
            leftSym = "《"
            rightSym = "》"
        Case 6  ' 英文尖括号
            leftSym = "<"
            rightSym = ">"
        Case 7  ' 中文方括号
            leftSym = "【"
            rightSym = "】"
        Case 8  ' 英文方括号
            leftSym = "["
            rightSym = "]"
        Case 9  ' 中文花括号「」
            leftSym = "「"
            rightSym = "」"
        Case 10 ' 英文花括号
            leftSym = "{"
            rightSym = "}"
        Case 11 ' 中文圆括号（）
            leftSym = "（"
            rightSym = "）"
        Case 12 ' 英文圆括号
            leftSym = "("
            rightSym = ")"
        Case Else
            Return False
    End Select
    Return True
End Function


' 我的函数
' 插入标点符号到当前活动窗口的焦点控件
Sub InsertChineseQuotes(i As Integer)
   Dim hWndFore As hWnd
   Dim hWndFocus As hWnd
   Dim tid As Long, pid As Long
   Dim selectedText As String
   Dim newText As String

   ' 获取前台窗口和焦点控件
   hWndFore = GetForegroundWindow()
   If hWndFore = 0 Then Exit Sub
   
   tid = GetWindowThreadProcessId(hWndFore, @pid)
   AttachThreadInput(GetCurrentThreadId(), tid, True)
   hWndFocus = GetFocus()
   AttachThreadInput(GetCurrentThreadId(), tid, False)
   
   If hWndFocus = 0 Then Exit Sub

   ' 1. 发送 Ctrl+C 复制选中文本
   FF_KeySend(hWndFocus, VK_CONTROL, KEY_DOWN)
   FF_KeySend(hWndFocus, Asc("C"), KEY_CLICK)
   FF_KeySend(hWndFocus, VK_CONTROL, KEY_UP)
   
   ' 稍等片刻让剪贴板更新（可调整延时）
   Sleep(50)
   
   ' 2. 获取剪贴板文本
   selectedText = FF_ClipboardGetText()
   
   ' 3. 构建新文本
   If selectedText <> "" Then
      newText = "“" & selectedText & "”"
   Else
      ' 如果没有选中文本，则只插入一对引号，光标放在中间
      newText = "“”"
   End If
   
   ' 4. 将新文本放入剪贴板
   FF_ClipboardSetText(newText)
   
   ' 5. 发送 Ctrl+V 粘贴
   FF_KeySend(hWndFocus, VK_CONTROL, KEY_DOWN)
   FF_KeySend(hWndFocus, Asc("V"), KEY_CLICK)
   FF_KeySend(hWndFocus, VK_CONTROL, KEY_UP)
   
End Sub
