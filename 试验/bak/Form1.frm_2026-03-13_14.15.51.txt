'[Form1.Command1]事件 : 单击
'hWndForm    当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'hWndControl 当前控件的句柄(也是窗口句柄，如果多开本窗口，必须 Me.控件名.hWndForm = hWndForm 后才可以执行后续操作本控件的代码 )
DimP ss As String 
ss =  "  16 , 17 , 18  "
Sub Form1_Command1_BN_Clicked(hWndForm As hWnd, hWndControl As hWnd)
   Label1.Caption = ss
   
   Dim s       As String
   Dim parts() As String
   Dim i       As Integer
   Dim r       As Integer, g As Integer, b As Integer
   Dim count   As Integer
   Dim token   As String
   
   ' 将所有逗号替换为空格，统一分隔符
   s = YF_Replace(ss, ",", " ")
   
   ' 按空格分割字符串
   vbSplit(s, " ", parts())
   
   count = 0
   For i = LBound(parts) To UBound(parts)
      token = Trim(parts(i))
      If Len(Trim(parts(i))) > 0 Then
         count += 1
         Select Case count
            Case 1
               r = Val(token) ' 使用标准 Val
            Case 2
               g = Val(token)
            Case 3
               b = Val(token)
               Exit For
         End Select
      End If
   Next
   
   ' 钳位到 0-255
   If r < 0 Then r = 0 Else If r > 255 Then r = 255
   If g < 0 Then g = 0 Else If g > 255 Then g = 255
   If b < 0 Then b = 0 Else If b > 255 Then b = 255
   
   ' 使用 & 连接字符串
   Label2.Caption= "#" + Right("0" & Hex(r), 2) + Right("0" & Hex(g), 2) + Right("0" & Hex(b), 2)
   
End Sub
