DimP prefix       As String
DimP suffix       As String
DimP serial       As String
DimP text_serial  As String
DimP num_serial   As Integer
DimP serial_width As Integer

' 格式化序列号函数(参数：数字，长度)
Function FormatSerial(num As Integer, wid As Integer) As String
    Dim limit As Integer
    Dim result As String
    
    limit = 10 ^ wid   ' 计算临界值，如 wid=3 则 limit=1000
    
    If num < limit Then
        ' 不足 wid 位，补前导零
        result = Right(String(wid, Asc("0")) & Trim(Str(num)), wid)
    Else
        ' 超过或等于 limit，直接显示数字（去除多余空格）
        result = Trim(Str(num))
    End If
    
    Return result
End Function

'[Form1.Command_Set]事件 : 单击
'hWndForm    当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'hWndControl 当前控件的句柄(也是窗口句柄，如果多开本窗口，必须 Me.控件名.hWndForm = hWndForm 后才可以执行后续操作本控件的代码 )
Sub Form1_Command_Set_BN_Clicked(hWndForm As hWnd, hWndControl As hWnd)
   Form2.Show(Form1.hWnd, True)
   serial_width = Len(serial)
   num_serial   = ValInt(serial)
   ' 将 num_serial 格式化为宽度为 serial_width 的字符串（保留位数）
   Dim serial_display As String
   Dim num_str        As String
   num_str = Trim(Str(num_serial)) ' 数值转字符串，去掉前导空格
   
   If Len(num_str) <= serial_width Then
      ' 不足宽度，前面补 '0'
      serial_display = Right(String(serial_width, "0") & num_str, serial_width)
   Else
      ' 超过宽度，直接显示实际数字（不补零）
      serial_display = num_str
   End If
   
   ' 4. 拼接最终显示信息
   Dim NowInfo As String
   NowInfo = prefix & serial_display & suffix
   
   ' 5. 显示到 Label_Show 上（假设你的标签控件名为 Label_Show）
   Label_Show.Caption = NowInfo
End Sub
