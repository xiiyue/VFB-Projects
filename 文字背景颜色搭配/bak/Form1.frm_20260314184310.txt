Sub RGB格式化(s As String, outrgb() As Integer)
   Dim parts() As String
   Dim i       As Integer, index As Integer
   Dim cleaned As String
   
   ' 1. 将所有逗号替换为空格，统一分隔符
   cleaned = YF_Replace(s, ",", " ")
   
   ' 2. 按空格分割成数组（连续空格会产生空字符串）
   vbSplit(cleaned, " ", parts())
   
   ' 3. 遍历 parts，提取有效数字存入 rgb 数组
   index = 0
   For i = 0 To UBound(parts)
      cleaned = Trim(parts(i))
      If Len(cleaned) > 0 Then
         outrgb(index) = Val(cleaned)
         index += 1
         If index >= 3 Then Exit For 
      End If
   Next i
End Sub
