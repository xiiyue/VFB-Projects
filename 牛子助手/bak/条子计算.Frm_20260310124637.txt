DimP stage(6) As String
stage(0) = "原木"
stage(1) = "青铜"
stage(2) = "白银"
stage(3) = "黄金"
stage(4) = "白金"
stage(5) = "秘银"
stage(6) = "钻石"
DimP stageData(6, 2) As Integer = {{1, 100, 4}, {101, 220, 4}, {221, 395, 5}, {396, 595, 5}, {596, 820, 5}, {821, 1120, 6}, {1121, 1450, 6}}
DimP selectstage As Integer = 0         ' 代表阶段选择的索引（0开始）
DimP selectsubstage As Integer = 0 ' 代表进度选择的数字 （1开始）       selectsubstage = 进度选择.ListIndex +1 之后清除
DimP barnum As Integer = 0 ' 当前阶段的当前进度下，界面里显示的条子量   barnum = ValInt(条子输入.Text) 之后清除
DimP mybar As Integer = 0 ' 目前获得的条子总量
DimP speed As Integer = 10   ' speed = ValInt(速度输入.Text) 之后清除
DimP 某阶段某进度的条子区间(5,1) As Integer     ' 第一项进度，第二项左右区间

Sub 计算区间(i As Integer)
   ' i 指 选择的阶段selectstage
   Dim 始条量     As Integer = stageData(i, 0)
   Dim 终条量     As Integer = stageData(i, 1)
   Dim 进度个数   As Integer = stageData(i, 2)
   Dim 进度的长度 As Integer
   Dim 左区间     As Integer
   Dim 右区间     As Integer
   Dim t          As Integer
   进度的长度 = (终条量 - 始条量 + 1) / 进度个数
   Erase 某阶段某进度的条子区间
   For t = 0 To 进度个数
      左区间 = Fix(始条量 + i *进度的长度)
      右区间 = Fix(始条量 + (i + 1) *进度的长度 -1)
      某阶段某进度的条子区间(t, 0) = 左区间
      某阶段某进度的条子区间(t, 1) = 右区间
   Next
 
End Sub



Sub 结果计算()
   Dim 当前进度 As Integer =  进度选择.ListIndex +1
   Dim 当前阶段进度的左区间 As Integer = 某阶段某进度的条子区间(当前进度, 0)
   Dim barall As Integer = 1450
   mybar = (当前阶段进度的左区间 -1) + barnum
   Dim 剩余总条子数 = barall - mybar
   
End Sub

   ' 初始化
   Dim i As Integer
   For i = LBound(stage) To UBound(stage)
      阶段选择.AddItem stage(i)
   Next i
   阶段选择.ListIndex = 0
   For i = 1 To 4
      进度选择.AddItem Str(i)
   Next
   进度选择.ListIndex = 0
End Sub

'[条子计算.阶段选择]事件 : 列表框中更改当前选择时
'hWndForm    当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)

Sub 条子计算_阶段选择_CBN_SelChange(hWndForm As hWnd, hWndControl As hWnd)
   Dim s = 阶段选择.ListIndex
   ' 刷新进度选择下拉框
   If s >= 0 And s <= 6 Then
      Dim 最大进度 As Integer = stageData(s, 2)
      ' 清空进度列表
      进度选择.Clear
      Dim i As Integer
      For i = 1 To 最大进度
         进度选择.AddItem Str(i)
      Next
      进度选择.ListIndex = 0
   End If
End Sub
