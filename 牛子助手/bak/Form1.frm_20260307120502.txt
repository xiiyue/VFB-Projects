'计时器初始值
'可以写成一行= {{7, 30}, {2, 10}, {9, 50}}
Dim Shared 初始倒计时(1 To 3, 1 To 2) As Integer = _
{ _
    {7, 30}, _ ' 初始倒计时(1,1) 和 (1,2)
    {2, 10}, _ ' 初始倒计时(2,1) 和 (2,2)
    {9, 50} _  ' 初始倒计时(3,1) 和 (3,2)
}

Dim Shared 固定总时间(1 To 3) As Double  '毫秒
Dim Shared 使用总时间(1 To 3) As Double  '毫秒
Dim Shared 开始点(1 To 3) As Double
Dim Shared 工作中(1 To 3) As Boolean '默认false

' 更新控件索引倒计时的显示标签
Sub 更新倒计时显示(i As Integer, 显示时间 As Double)
    Dim 分钟 As Integer, 秒 As Integer
    If 显示时间 <= 0 Then
        分钟 = 0
        秒 = 0
    Else
        分钟 = Int(显示时间 / 60000)
        秒 = Int((显示时间 Mod 60000) / 1000)
     End If
     倒计时(i).Caption = Format(分钟, "00") & ":" & Format(秒, "00")
End Sub

' 重置倒计时
Sub 重置倒计时(i As Integer)
   工作中(i) = False 
   启动(i-1).Caption = "开始"
   启动(i -1).BackColor = RGB(255, 255, 255)
   开始点(i) = 0
   使用总时间(i) = 固定总时间(i)
End Sub



'[Form1]事件 : 窗口完全显示后。
'hWndForm  当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'UserData  来自显示窗口最后1个参数，例： Form2.Show(父窗口句柄,模式,UserData)
Sub Form1_Shown(hWndForm As hWnd, UserData As Integer)
   Dim i As Integer
   For i = 1 To 3
    ' 初始化总时间，分钟秒钟转为毫秒
    固定总时间(i) = 初始倒计时(i, 1) * 60 * 1000 + 初始倒计时(i, 2) * 1000
    使用总时间(i) = 固定总时间(i)
    ' 初始化倒计时显示
    倒计时(i-1).Caption = Format(初始倒计时(i , 1), "00") & ":" & Format(初始倒计时(i, 2), "00")
   Next i 

End Sub

'[Form1.Timer1]事件 : 定时器
'hWndForm  当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
''         本控件为功能控件，就是无窗口，无显示，只有功能。如果多开本窗口，必须 Me.控件名.hWndForm = hWndForm 后才可以执行后续操作本控件的代码 
'wTimerID  定时器ID，区分不同定时器的编号
Sub Form1_Timer1_WM_Timer(hWndForm As hWnd, wTimerID As Long)
   Dim i As Integer
   Dim 消耗时间 As Double            ' 单位秒
   Dim 显示时间 As Double            ' 单位毫秒
   
   For i = 1 To 3
      If 工作中(i) = True Then
         消耗时间 = Timer - 开始点(i)
         ' 跨夜补偿
         If 消耗时间 < 0 Then 消耗时间 = 消耗时间 + 86400
         
         显示时间 = 使用总时间(i) - 消耗时间 * 1000
         
         ' 倒计时正常结束
         If 显示时间 <= 0 Then
            重置倒计时(i)
         End If
         更新倒计时显示(i-1,显示时间)
      End If  
   Next i

   ' 如果所有倒计时都停止，则关闭定时器
   If 工作中(1) = False And 工作中(2) = False And 工作中(3) = False Then
        Timer1.Enabled = False
   End If

End Sub

'[Form1.启动]事件 : 单击
'hWndForm    当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'hWndControl 当前控件的句柄(也是窗口句柄，如果多开本窗口，必须 Me.控件名.hWndForm = hWndForm 后才可以执行后续操作本控件的代码 )
Sub Form1_启动_BN_Clicked(ControlIndex As Long, hWndForm As hWnd, hWndControl As hWnd)
   ' 切换工作状态
   If 工作中(ControlIndex + 1) = False Then
      工作中(ControlIndex + 1) = True
      启动(ControlIndex).Caption = "重置"
      启动(ControlIndex).BackColor = RGB(255, 0, 0) ' 红色
      开始点(ControlIndex + 1) = Timer
   Else
      重置倒计时(ControlIndex + 1)
      更新倒计时显示(ControlIndex,使用总时间(ControlIndex + 1))
   End If
   ' 只要有一个倒计时在工作，就开启定时器
   Timer1.Enabled = (工作中(1) Or 工作中(2) Or 工作中(3))
End Sub
