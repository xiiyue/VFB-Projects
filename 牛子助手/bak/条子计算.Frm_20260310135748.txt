DimP stage(6) As String
stage(0) = "原木"
stage(1) = "青铜"
stage(2) = "白银"
stage(3) = "黄金"
stage(4) = "白金"
stage(5) = "秘银"
stage(6) = "钻石"
DimP stageData(6, 2) As Integer = {{1, 100, 4}, {101, 220, 4}, {221, 395, 5}, {396, 595, 5}, {596, 820, 5}, {821, 1120, 6}, {1121, 1450, 6}}


'[条子计算]事件 : 窗口完全显示后。
'hWndForm  当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'UserData  来自显示窗口最后1个参数，例： Form2.Show(父窗口句柄,模式,UserData)
Sub 条子计算_Shown(hWndForm As hWnd, UserData As Integer)
   ' 初始化
   Dim i As Integer
   For i = LBound(stage) To UBound(stage)
      阶段选择.AddItem stage(i)
   Next i
   阶段选择.ListIndex = 0
   For i = 1 To 4
      进度选择.AddItem Str(i)
   Next
   进度选择.ListIndex = 0
End Sub

'[条子计算.阶段选择]事件 : 列表框中更改当前选择时
'hWndForm    当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'hWndControl 当前控件的句柄(也是窗口句柄，如果多开本窗口，必须 Me.控件名.hWndForm = hWndForm 后才可以执行后续操作本控件的代码 )
Sub 条子计算_阶段选择_CBN_SelChange(hWndForm As hWnd, hWndControl As hWnd)
   Dim s = 阶段选择.ListIndex
   ' 刷新进度选择下拉框
   If s >= 0 And s <= 6 Then
      Dim 最大进度 As Integer = stageData(s, 2)
      ' 清空进度列表
      进度选择.Clear
      Dim i As Integer
      For i = 1 To 最大进度
         进度选择.AddItem Str(i)
      Next
      进度选择.ListIndex = 0
   End If
End Sub


'[条子计算.计算按钮]事件 : 单击
'hWndForm    当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'hWndControl 当前控件的句柄(也是窗口句柄，如果多开本窗口，必须 Me.控件名.hWndForm = hWndForm 后才可以执行后续操作本控件的代码 )
Sub 条子计算_计算按钮_BN_Clicked(hWndForm As hWnd, hWndControl As hWnd)
   Dim 当前阶段    As Integer = 阶段选择.ListIndex
   Dim 当前进度    As Integer = 进度选择.ListIndex + 1
   Dim mybar       As Integer = 0 ' 目前获得的条子总量
   Dim barnum      As Integer = ValInt(条子输入.Text) '额外条子量
   Dim speed       As Integer = ValInt(速度输入.Text)
   Dim 当前区间左 As Integer

   
   '计算区间
   Dim 始条量     As Integer = stageData(当前阶段, 0)
   Dim 终条量     As Integer = stageData(当前阶段, 1)
   Dim 进度个数   As Integer = stageData(当前阶段, 2)
   Dim 进度的长度 As Integer = (终条量 - 始条量 + 1) / 进度个数
   当前区间左 = Fix(始条量 + 当前进度 *进度的长度)
   '当前区间右 = Fix(始条量 + (当前进度 + 1) *进度的长度 -1)
   
   mybar = (当前区间(0) -1) + barnum

   
   ' 计算全部完成需要时间
    计算时间并展示(1450, mybar, speed,1)
   
   ' 计算本阶段完成需要时间
    计算时间并展示(终条量, mybar, speed,0)
End Sub


Sub 计算时间并展示(allbar As Integer, nowbar As Integer, speed As Integer, idx As Integer)
   Dim 剩余条子数      As Integer
   Dim 剩余多少个5分钟 As Integer
   Dim 所需时间(1)     As Integer
   剩余条子数          = allbar - nowbar
   剩余多少个5分钟     = ceil(剩余条子数 / speed)
   所需时间(0)         = (剩余多少个5分钟 * 5) \ 60   ' hour
   所需时间(1)         = (剩余多少个5分钟 * 5) Mod 60 'min
   展示时间(idx).Caption = Format(所需时间(0), "00") & "时"& Format(所需时间(1), "00") & "分"
End Sub
