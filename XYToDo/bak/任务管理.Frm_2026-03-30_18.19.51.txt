'[任务管理]事件 : 窗口完全显示后。
'hWndForm  当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'UserData  来自显示窗口最后1个参数，例： Form2.Show(父窗口句柄,模式,UserData)
Sub 任务管理_Shown(hWndForm As hWnd, UserData As Integer)
   ' 初始化 全任务List
   Dim i        As Integer
   Dim itemData As Long
   List_全任务列表.Clear ' 清空原有内容
   List_全任务列表.AddItem "=====周期任务=====", -1
   For i = LBound(periodicTasks) To UBound(periodicTasks)
      itemData = (1 Shl 16) Or i ' 高16位 保存 任务类型； 低16位 保存该任务的数组索引
      List_全任务列表.AddItem periodicTasks(i).Name, itemData
   Next i
   List_全任务列表.AddItem "=====普通任务=====", -1
   For i = LBound(regularTasks) To UBound(regularTasks)
      itemData = (2 Shl 16) Or i
      List_全任务列表.AddItem regularTasks(i).Name, itemData
   Next i
End Sub

'[任务管理.List_全任务列表]事件 : 选择了列表
'hWndForm    当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'hWndControl 当前控件的句柄(也是窗口句柄，如果多开本窗口，必须 Me.控件名.hWndForm = hWndForm 后才可以执行后续操作本控件的代码 )
Sub 任务管理_List_全任务列表_LBN_SelChange(hWndForm As hWnd, hWndControl As hWnd)
   Dim idx As Integer = List_全任务列表.ListIndex
   Dim i As Integer
   Dim subCount As Integer 
   If idx < 0 Then Exit Sub
   
   
   Dim itemData As Long = List_全任务列表.ItemData(idx)
   If itemData = -1 Then Exit Sub ' 标题行，忽略
   
   Dim taskType  As Integer = (itemData Shr 16) And &HFFFF ' 高16位，区分周期还是普通
   Dim taskIndex As Integer = itemData          And &HFFFF ' 低16位，任务数组索引
   
   ' 重置单选按钮
   Option1.Value = False
   Option2.Value = False
   Option3.Value = False
   Option4.Value = False
   Option5.Value = False
   
   ' 清空子任务列表
   List_子任务.Clear
   
   Select Case taskType
      Case 1 ' 周期任务
         Dim ptask As PeriodicTask = periodicTasks(taskIndex)
         ' 任务名编辑框
         Text_任务名.Text = ptask.Name
         ' 时间选择器
         DateTimePicker1.DateTime = AfxVariantTimeToSystemTime(ptask.nextPeriod)
         ' 周期选择
         Select Case ptask.PeriodType
            Case 1 ' 日
               Option2.Value = True
            Case 2 ' 周
               Option3.Value = True
            Case 3 ' 月
               Option4.Value = True
            Case 4 ' 年
               Option5.Value = True
            Case Else ' 0 或其他，按普通任务处理
               Option1.Value = True
         End Select
         ' 填充子任务列表
         subCount = Val(ptask.subtasks(0, 1))  ' 子任务数量（字符串转数值）
         For i = 1 To subCount
            List_子任务.AddItem ptask.subtasks(i, 0)  ' 添加任务名（列0）
         Next i
   
      Case 2 ' 普通任务
         Dim rtask As RegularTask = regularTasks(taskIndex)
         Text_任务名.Text         = rtask.Name
         DateTimePicker1.DateTime = AfxVariantTimeToSystemTime(rtask.taskTime)
         Option1.Value = True
         ' 填充子任务列表
         subCount = Val(rtask.subtasks(0, 1))  ' 子任务数量（字符串转数值）
         For i = 1 To subCount
            List_子任务.AddItem rtask.subtasks(i, 0)  ' 添加任务名（列0）
         Next i
   End Select
  
End Sub

'[任务管理.YFbunGroup1]事件 : 点击了某个按钮
'hWndForm  当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'nID       按钮号，鼠标点击了某个按钮，就是点击后释放鼠标时触发
Function 任务管理_YFbunGroup1_Command(hWndForm As hWnd, hWndControl As hWnd, nID As Long ) As Long
   Dim idx As Integer = List_全任务列表.ListIndex
   Dim itemData As Long = List_全任务列表.ItemData(idx)
   If itemData = -1 Then Exit Sub ' 标题行，忽略
   
   Dim taskType  As Integer = (itemData Shr 16) And &HFFFF ' 高16位，任务类型，1是periodicTasks()，2是regularTasks()
   Dim taskIndex As Integer = itemData And &HFFFF ' 低16位，任务数组索引，精确定位到所选的任务，periodicTasks(taskIndex)或regularTasks(taskIndex)
   
   Dim subtask_name As String
   subtask_name = Text_子任务.Text
   
   Select Case nID
      Case 任务管理_YFbunGroup1_addsubtask ' 添加子任务

      Case 任务管理_YFbunGroup1_editsubtask ' 修改子任务

      Case 任务管理_YFbunGroup1_delsubtask ' 删除子任务

      Case 任务管理_YFbunGroup1_moveup ' 上移

      Case 任务管理_YFbunGroup1_movedown ' 下移

   End Select

   Function = TRUE '  如果你处理了事件，就返回 Return TRUE ,控件就不进行默认处理了。

End Function
