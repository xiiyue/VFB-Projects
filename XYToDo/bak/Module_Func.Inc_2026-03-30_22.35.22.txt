' 本模块是操作任务数组以及自定义功能
#include Once "Afx/AfxTime.inc"
' ==========================================
'刷新子任务列表（根据任务类型和索引）
' ==========================================
Sub RefreshSubtaskList(taskType As Integer, taskIndex As Integer)
   任务管理.List_子任务.Clear
   Dim count As Integer = 0
   If taskType = 1 Then
      count = Val(periodicTasks(taskIndex).subtasks(0, 1))
      For i As Integer = 1 To count
         任务管理.List_子任务.AddItem periodicTasks(taskIndex).subtasks(i, 0)
      Next
   Else
      count = Val(regularTasks(taskIndex).subtasks(0, 1))
      For i As Integer = 1 To count
         任务管理.List_子任务.AddItem regularTasks(taskIndex).subtasks(i, 0)
      Next
   End If
End Sub
' ==========================================
' 刷新全任务列表
' ==========================================
Sub RefreshFullTaskList()
   Dim i        As Integer
   Dim itemData As Long
   任务管理.List_全任务列表.Clear ' 清空原有内容
   任务管理.List_全任务列表.AddItem "=====普通任务=====", -1
   For i = LBound(regularTasks) To UBound(regularTasks)
      itemData = (2 Shl 16) Or i
      任务管理.List_全任务列表.AddItem regularTasks(i).Name, itemData
   Next i
   任务管理.List_全任务列表.AddItem "=====周期任务=====", -1
   For i = LBound(periodicTasks) To UBound(periodicTasks)
      itemData = (1 Shl 16) Or i ' 高16位 保存 任务类型； 低16位 保存该任务的数组索引
      任务管理.List_全任务列表.AddItem periodicTasks(i).Name, itemData
   Next i
End Sub
' ==========================================
' 计算下期执行时间
' ==========================================
Function CalculateNextPeriod(currentPeriod As Double, periodType As Integer) As Double
    Dim sysTime As SYSTEMTIME
    AfxVariantTimeToSystemTime(currentPeriod, sysTime)
    
    Dim newYear As Integer = sysTime.wYear
    Dim newMonth As Integer = sysTime.wMonth
    Dim newDay As Integer = sysTime.wDay
    Dim newHour As Integer = sysTime.wHour
    Dim newMinute As Integer = sysTime.wMinute
    Dim newSecond As Integer = sysTime.wSecond
    
    Select Case periodType
        Case 1 ' 日任务
            newDay += 1
            AdjustDate(newYear, newMonth, newDay)
            
        Case 2 ' 周任务
            newDay += 7
            AdjustDate(newYear, newMonth, newDay)
            
        Case 3 ' 月任务
            newMonth += 1
            If newMonth > 12 Then
                newMonth = 1
                newYear += 1
            End If
            ' 如果原日期大于新月份的天数，则取新月份最后一天
            Dim daysInNewMonth As Integer = GetDaysInMonth(newYear, newMonth)
            If newDay > daysInNewMonth Then
                newDay = daysInNewMonth
            End If
            
        Case 4 ' 年任务
            newYear += 1
            ' 处理闰年2月29日的情况
            If newMonth = 2 And newDay = 29 Then
                Dim daysInFeb As Integer = GetDaysInMonth(newYear, 2)
                If daysInFeb < 29 Then
                    newDay = daysInFeb
                End If
            End If
    End Select
    
    ' 组合新的 SYSTEMTIME
    Dim newSysTime As SYSTEMTIME
    newSysTime.wYear = newYear
    newSysTime.wMonth = newMonth
    newSysTime.wDay = newDay
    newSysTime.wHour = newHour
    newSysTime.wMinute = newMinute
    newSysTime.wSecond = newSecond
    newSysTime.wMilliseconds = 0
    
    ' 转换回 OLE 日期序列号并返回
    Return AfxSystemTimeToVariantTime(newSysTime)
End Function

' 辅助程序：调整日期（处理日进位到月、月进位到年）
Sub AdjustDate(ByRef year As Integer, ByRef month As Integer, ByRef day As Integer)
    While day > GetDaysInMonth(year, month)
        day -= GetDaysInMonth(year, month)
        month += 1
        If month > 12 Then
            month = 1
            year += 1
        End If
    Wend
End Sub

' 辅助函数：获取某年某月的天数
Function GetDaysInMonth(year As Integer, month As Integer) As Integer
    Select Case month
        Case 1, 3, 5, 7, 8, 10, 12
            Return 31
        Case 4, 6, 9, 11
            Return 30
        Case 2
            If (year Mod 400 = 0) Or (year Mod 4 = 0 And year Mod 100 <> 0) Then
                Return 29
            Else
                Return 28
            End If
    End Select
End Function
' ==========================================
' 根据任务类型和索引删除任务
' ==========================================
Function DeleteTaskByTypeAndIndex(taskType As Integer, taskIndex As Integer) As Boolean
    If taskType = 1 Then   ' 周期任务
        Dim cnt As Integer = UBound(periodicTasks)
        If taskIndex < 0 Or taskIndex > cnt Then Return False
        ' 将待删除元素之后的元素向前移动
        For i As Integer = taskIndex To cnt - 1
            periodicTasks(i) = periodicTasks(i + 1)
        Next
        ' 缩小数组
        If cnt > 0 Then
            ReDim Preserve periodicTasks(0 To cnt - 1)
        Else
            ReDim periodicTasks(0 To -1)   ' 清空数组
        End If
        Return True
    ElseIf taskType = 2 Then   ' 普通任务
        Dim cnt As Integer = UBound(regularTasks)
        If taskIndex < 0 Or taskIndex > cnt Then Return False
        For i As Integer = taskIndex To cnt - 1
            regularTasks(i) = regularTasks(i + 1)
        Next
        If cnt > 0 Then
            ReDim Preserve regularTasks(0 To cnt - 1)
        Else
            ReDim regularTasks(0 To -1)
        End If
        Return True
    End If
    Return False
End Function
