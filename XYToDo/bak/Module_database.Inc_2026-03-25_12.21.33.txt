
' 加载周期任务
Function LoadPeriodicTasks(periodicTasks() As PeriodicTask, filePath As String) As Long
    Dim hFile As Long = FreeFile
    Dim fileContent As String
    Dim jsonRoot As cJSON Ptr
    Dim jsonItem As cJSON Ptr
    Dim taskCount As Long
    Dim i As Long
    
    ' 尝试打开文件
    If Not FileExists(filePath) Then
        ' 文件不存在，返回空数组
        ReDim periodicTasks(0)
        Return 0
    End If
    
    Open filePath For Binary As #hFile
    fileContent = Space(Lof(hFile))
    Get #hFile, , fileContent
    Close #hFile
    
    ' 解析 JSON
    jsonRoot = cJSON_Parse(fileContent)
    If jsonRoot = 0 Then
        ' 解析失败，返回空数组
        ReDim periodicTasks(0)
        Return 0
    End If
    
    ' 获取数组大小
    taskCount = cJSON_GetArraySize(jsonRoot)
    ReDim periodicTasks(taskCount - 1)
    
    ' 遍历数组
    For i = 0 To taskCount - 1
        jsonItem = cJSON_GetArrayItem(jsonRoot, i)
        With periodicTasks(i)
            .Name = *cJSON_GetObjectItem(jsonItem, "name")->valuestring
            .createTime = cJSON_GetObjectItem(jsonItem, "createTime")->valuedouble
            .PeriodType = cJSON_GetObjectItem(jsonItem, "periodType")->valueint
            .currentPeriod = cJSON_GetObjectItem(jsonItem, "currentPeriod")->valuedouble
            .nextPeriod = cJSON_GetObjectItem(jsonItem, "nextPeriod")->valuedouble
            .completed = cJSON_GetObjectItem(jsonItem, "completed")->valueint <> 0
            .completionCount = cJSON_GetObjectItem(jsonItem, "completionCount")->valueint
            .onList = cJSON_GetObjectItem(jsonItem, "onList")->valueint <> 0
            .tempStatus = cJSON_GetObjectItem(jsonItem, "tempStatus")->valueint <> 0
            .subtasks = *cJSON_GetObjectItem(jsonItem, "subtasks")->valuestring
        End With
    Next i
    
    cJSON_Delete(jsonRoot)
    Return taskCount
End Function


' 保存周期任务
Sub SavePeriodicTasks(periodicTasks() As PeriodicTask, filePath As String)
    Dim hFile As Long = FreeFile
    Dim jsonRoot As cJSON Ptr
    Dim jsonItem As cJSON Ptr
    Dim jsonStr As String
    Dim i As Long
    
    ' 创建 JSON 数组
    jsonRoot = cJSON_CreateArray()
    
    ' 遍历任务数组
    For i = LBound(periodicTasks) To UBound(periodicTasks)
        jsonItem = cJSON_CreateObject()
        With periodicTasks(i)
            cJSON_AddStringToObject(jsonItem, "name", .Name)
            cJSON_AddNumberToObject(jsonItem, "createTime", .createTime)
            cJSON_AddNumberToObject(jsonItem, "periodType", .PeriodType)
            cJSON_AddNumberToObject(jsonItem, "currentPeriod", .currentPeriod)
            cJSON_AddNumberToObject(jsonItem, "nextPeriod", .nextPeriod)
            cJSON_AddBoolToObject(jsonItem, "completed", .completed)
            cJSON_AddNumberToObject(jsonItem, "completionCount", .completionCount)
            cJSON_AddBoolToObject(jsonItem, "onList", .onList)
            cJSON_AddBoolToObject(jsonItem, "tempStatus", .tempStatus)
            cJSON_AddStringToObject(jsonItem, "subtasks", .subtasks)
        End With
        cJSON_AddItemToArray(jsonRoot, jsonItem)
    Next i
    
    ' 转换为字符串并保存
    jsonStr = *cJSON_Print(jsonRoot)
    Open filePath For Output As #hFile
    Print #hFile, jsonStr
    Close #hFile
    
    ' 释放内存
    cJSON_Delete(jsonRoot)
    cJSON_free(jsonStr)
End Sub
