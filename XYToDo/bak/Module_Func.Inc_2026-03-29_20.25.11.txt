' 本模块是操作任务数组以及自定义功能
#include Once "Afx/AfxTime.inc"


' 解析子任务字符串，填充到二维数组 subItems(行, 列)
' 列0: 子任务名称，列1: 状态（"0" 或 "1"）
Sub ParseSubtasks(subtasksStr As String, subItems(,) As String)
    ' 清空/重置数组
    If Len(subtasksStr) = 0 Then
        ReDim subItems(-1, -1)   ' 空数组
        Exit Sub
    End If
    
    Dim parts() As String = Split(subtasksStr, "|")
    Dim count As Integer = UBound(parts) - LBound(parts) + 1
    ReDim subItems(count - 1, 1)   ' 二维数组，行数=子任务个数，列数=2
    
    For i As Integer = LBound(parts) To UBound(parts)
        Dim subPart() As String = Split(parts(i), ":")
        If UBound(subPart) >= 0 Then
            subItems(i, 0) = subPart(0)                 ' 子任务名称
            If UBound(subPart) >= 1 Then
                subItems(i, 1) = subPart(1)             ' 状态（"0"/"1"）
            Else
                subItems(i, 1) = "0"
            End If
        Else
            subItems(i, 0) = ""
            subItems(i, 1) = "0"
        End If
    Next i
End Sub
