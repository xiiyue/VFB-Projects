Sub RefreshTodayForm()
    Dim i As Long
    Dim j As Long
    Dim taskIdx As Long
    Dim hasSub As Long
    Dim subCount As Long
    
    Dim rootPeriodic As YFTreeViewData Ptr
    Dim rootRegular As YFTreeViewData Ptr
    Dim rootDeferred As YFTreeViewData Ptr
    Dim taskNode As YFTreeViewData Ptr
    Dim subNode As YFTreeViewData Ptr
    Dim dataValue As Long
    
    ' 清空两个树形视图
    YFTreeView_今日列表.DeleteAllItems
    YFTreeView_副列表.DeleteAllItems
    
    ' ========== 今日列表：周期任务 ==========
    If UBound(periodicOnList) >= 0 Then
        For i = 0 To UBound(periodicOnList)
            taskIdx = periodicOnList(i)
            If taskIdx >= 0 And taskIdx <= UBound(periodicTasks) Then
                With periodicTasks(taskIdx)
                    ' 编码 DataValue: 高16位=1(周期), 低16位=taskIdx
                    dataValue = (1 Shl 16) Or taskIdx
                    ' 添加主任务节点
                    taskNode = YFTreeView_今日列表.AddItem(0, .Name, dataValue, 0, 0, 0, -1, -1, .tempStatus)
                    ' 检查是否有子任务
                    hasSub = Val(.subtasks(0, 0))
                    If hasSub = 1 Then
                        subCount = Val(.subtasks(0, 1))
                        For j = 1 To subCount
                            ' 子任务 DataValue 设为 0
                            subNode = YFTreeView_今日列表.AddItem(taskNode, .subtasks(j, 0), 0, 0, 0, 0, -1, -1, Val(.subtasks(j, 1)))
                        Next
                    End If
                End With
            End If
        Next
    End If
    
    ' ========== 副列表 ==========
    rootRegular = YFTreeView_副列表.AddItem(0, "=====普通任务=====", -1, 0, 0, 0, -1, -1, 0)
    ' 添加普通任务
    If UBound(regularOnList) >= 0 Then
        For i = 0 To UBound(regularOnList)
            taskIdx = regularOnList(i)
            If taskIdx >= 0 And taskIdx <= UBound(regularTasks) Then
                With regularTasks(taskIdx)
                    dataValue = (2 Shl 16) Or taskIdx
                    taskNode = YFTreeView_副列表.AddItem(rootRegular, .name, dataValue, 0, 0, 0, -1, -1, .tempStatus)
                    hasSub = Val(.subtasks(0, 0))
                    If hasSub = 1 Then
                        subCount = Val(.subtasks(0, 1))
                        For j = 1 To subCount
                            subNode = YFTreeView_副列表.AddItem(taskNode, .subtasks(j, 0), 0, 0, 0, 0, -1, -1, Val(.subtasks(j, 1)))
                        Next
                    End If
                End With
            End If
        Next
    End If
    
    ' ----- 延期任务标题 -----
    rootDeferred = YFTreeView_副列表.AddItem(0, "=====延期任务=====", -1, 0, 0, 0, -1, -1, 0)
    ' 添加所有延期任务
    If UBound(deferredTasks) >= 0 Then
        For i = 0 To UBound(deferredTasks)
            With deferredTasks(i)
                dataValue = (3 Shl 16) Or i
                taskNode = YFTreeView_副列表.AddItem(rootDeferred, .name, dataValue, 0, 0, 0, -1, -1, .tempStatus)
                hasSub = Val(.subtasks(0, 0))
                If hasSub = 1 Then
                    subCount = Val(.subtasks(0, 1))
                    For j = 1 To subCount
                        subNode = YFTreeView_副列表.AddItem(taskNode, .subtasks(j, 0), 0, 0, 0, 0, -1, -1, Val(.subtasks(j, 1)))
                    Next
                End If
            End With
        Next
    End If
End Sub
