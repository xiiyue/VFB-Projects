' ==========================================
' 刷新全任务列表
' ==========================================
Sub RefreshFullTaskList()
   Dim i        As Integer
   Dim itemData As Long
   List_全任务列表.Clear ' 清空原有内容
   List_全任务列表.AddItem "=====普通任务=====", -1
   For i = LBound(regularTasks) To UBound(regularTasks)
      itemData = (2 Shl 16) Or i
      List_全任务列表.AddItem regularTasks(i).Name, itemData
   Next i
   List_全任务列表.AddItem "=====周期任务=====", -1
   For i = LBound(periodicTasks) To UBound(periodicTasks)
      itemData = (1 Shl 16) Or i ' 高16位 保存 任务类型； 低16位 保存该任务的数组索引
      List_全任务列表.AddItem periodicTasks(i).Name, itemData
   Next i
End Sub

' ==========================================
'刷新子任务列表（根据任务类型和索引）
' ==========================================
Sub RefreshSubtaskList(taskType As Integer, taskIndex As Integer)
   List_子任务.Clear
   Dim count As Integer = 0
   If taskType = 1 Then
      count = Val(periodicTasks(taskIndex).subtasks(0, 1))
      For i As Integer = 1 To count
         List_子任务.AddItem periodicTasks(taskIndex).subtasks(i, 0)
      Next
   Else
      count = Val(regularTasks(taskIndex).subtasks(0, 1))
      For i As Integer = 1 To count
         List_子任务.AddItem regularTasks(taskIndex).subtasks(i, 0)
      Next
   End If
End Sub

' ==========================================
' 修改任务
' 参数：taskType - 原任务类型 (1=周期, 2=普通, 3=延期)
'       taskIndex - 在原数组中的索引
' ==========================================
Sub ModifyTask(taskType As Integer, taskIndex As Integer)
    Dim newName As String
    Dim newDateTime As Double
    Dim newPeriodType As Integer
    Dim newTaskType As Integer
    
    ' 1. 获取界面数据
    newName = Trim(Text_任务名.Text)
    If newName = "" Then
        MsgBox("任务名不能为空！")
        Exit Sub
    End If
    
    ' 获取日期时间（OLE 日期格式）
    Dim sysTime As SYSTEMTIME
    sysTime = DateTimePicker1.DateTime
    newDateTime = SetTimeTo4AM(sysTime)   ' 时间调整到凌晨4点
    
    ' 确定周期类型
    If Option1.Value = True Then   ' 无周期 => 普通任务
        newPeriodType = 0
    ElseIf Option2.Value = True Then
        newPeriodType = 1   ' 日
    ElseIf Option3.Value = True Then
        newPeriodType = 2   ' 周
    ElseIf Option4.Value = True Then
        newPeriodType = 3   ' 月
    ElseIf Option5.Value = True Then
        newPeriodType = 4   ' 年
    Else
        newPeriodType = 0   ' 默认无周期
    End If
    
    ' 新任务类型：周期类型为0 => 普通任务(2)，否则周期任务(1)
    If newPeriodType = 0 Then
        newTaskType = 2   ' 普通任务
    Else
        newTaskType = 1   ' 周期任务
    End If
    
    ' 2. 根据任务类型是否变化执行不同操作
    If taskType = newTaskType Then
        ' 类型未变，直接修改原任务
        If taskType = 1 Then   ' 周期任务
            With periodicTasks(taskIndex)
                .Name = newName
                .PeriodType = newPeriodType
                .currentPeriod = newDateTime
                .nextPeriod = CalculateNextPeriod(.currentPeriod, .PeriodType)
            End With
        ElseIf taskType = 2 Then   ' 普通任务
            With regularTasks(taskIndex)
                .Name = newName
                .taskTime = newDateTime
            End With
        End If
    Else
        ' 类型变化，需要创建新任务并复制子任务，然后删除原任务
        If newTaskType = 1 Then
            ' 创建新的周期任务
            Dim newPeriodic As PeriodicTask
            newPeriodic.id = GenerateTaskId()
            newPeriodic.Name = newName
            newPeriodic.createTime = regularTasks(taskIndex).createTime
            newPeriodic.PeriodType = newPeriodType
            newPeriodic.currentPeriod = newDateTime
            newPeriodic.nextPeriod = CalculateNextPeriod(newDateTime, newPeriodType)
            newPeriodic.completed = False
            newPeriodic.completionCount = 0
            newPeriodic.onList = False
            newPeriodic.tempStatus = False
            
            ' 复制子任务：从原任务中复制
            CopySubtasks(regularTasks(taskIndex).subtasks(), newPeriodic.subtasks())
            
            ' 添加到周期任务数组末尾
            Dim newIdx As Long = UBound(periodicTasks) + 1
            ReDim Preserve periodicTasks(newIdx)
            periodicTasks(newIdx) = newPeriodic
            
        ElseIf newTaskType = 2 Then
            ' 创建新的普通任务
            Dim newRegular As RegularTask
            newRegular.name = newName
            newRegular.createTime = periodicTasks(taskIndex).createTime
            newRegular.taskTime = newDateTime
            newRegular.onList = False
            newRegular.tempStatus = False
            
            ' 复制子任务
            CopySubtasks(periodicTasks(taskIndex).subtasks(), newRegular.subtasks())
            
            ' 添加到普通任务数组末尾
            Dim newIdx As Long = UBound(regularTasks) + 1
            ReDim Preserve regularTasks(newIdx)
            regularTasks(newIdx) = newRegular
        End If
        
        ' 删除原任务
        DeleteTask(taskType, taskIndex)
    End If
    MsgBox("修改成功！")
End Sub

'[任务管理.List_全任务列表]事件 : 选择了列表
'hWndForm    当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'hWndControl 当前控件的句柄(也是窗口句柄，如果多开本窗口，必须 Me.控件名.hWndForm = hWndForm 后才可以执行后续操作本控件的代码 )
Sub 任务管理_List_全任务列表_LBN_SelChange(hWndForm As hWnd, hWndControl As hWnd)
   Dim idx As Integer = List_全任务列表.ListIndex
   Dim i As Integer
   Dim subCount As Integer 
   If idx < 0 Then Exit Sub
   
   
   Dim itemData As Long = List_全任务列表.ItemData(idx)
   If itemData = -1 Then Exit Sub ' 标题行，忽略
   
   Dim taskType  As Integer = (itemData Shr 16) And &HFFFF ' 高16位，区分周期还是普通
   Dim taskIndex As Integer = itemData          And &HFFFF ' 低16位，任务数组索引
   
   ' 重置单选按钮
   Option1.Value = False
   Option2.Value = False
   Option3.Value = False
   Option4.Value = False
   Option5.Value = False
   
   ' 清空子任务列表
   List_子任务.Clear
   
   Select Case taskType
      Case 1 ' 周期任务
         Dim ptask As PeriodicTask = periodicTasks(taskIndex)
         ' 任务名编辑框
         Text_任务名.Text = ptask.Name
         ' 时间选择器
         DateTimePicker1.DateTime = AfxVariantTimeToSystemTime(ptask.nextPeriod)
         ' 周期选择
         Select Case ptask.PeriodType
            Case 1 ' 日
               Option2.Value = True
            Case 2 ' 周
               Option3.Value = True
            Case 3 ' 月
               Option4.Value = True
            Case 4 ' 年
               Option5.Value = True
            Case Else ' 0 或其他，按普通任务处理
               Option1.Value = True
         End Select
         ' 填充子任务列表
         subCount = Val(ptask.subtasks(0, 1))  ' 子任务数量（字符串转数值）
         For i = 1 To subCount
            List_子任务.AddItem ptask.subtasks(i, 0)  ' 添加任务名（列0）
         Next i
   
      Case 2 ' 普通任务
         Dim rtask As RegularTask = regularTasks(taskIndex)
         Text_任务名.Text         = rtask.Name
         DateTimePicker1.DateTime = AfxVariantTimeToSystemTime(rtask.taskTime)
         Option1.Value = True
         ' 填充子任务列表
         subCount = Val(rtask.subtasks(0, 1))  ' 子任务数量（字符串转数值）
         For i = 1 To subCount
            List_子任务.AddItem rtask.subtasks(i, 0)  ' 添加任务名（列0）
         Next i
   End Select
  
End Sub

'[任务管理.YFbunGroup1]事件 : 点击了某个按钮
'hWndForm  当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'nID       按钮号，鼠标点击了某个按钮，就是点击后释放鼠标时触发
Function 任务管理_YFbunGroup1_Command(hWndForm As hWnd, hWndControl As hWnd, nID As Long) As Long
   Dim idx      As Integer = List_全任务列表.ListIndex
   Dim itemData As Long    = List_全任务列表.ItemData(idx)
   If itemData = -1 Then Exit Function ' 标题行，忽略
   
   Dim taskType  As Integer = (itemData Shr 16) And &HFFFF ' 高16位，任务类型，1是periodicTasks()，2是regularTasks()
   Dim taskIndex As Integer = itemData          And &HFFFF ' 低16位，任务数组索引，精确定位到所选的任务，periodicTasks(taskIndex)或regularTasks(taskIndex)
   
   Dim subtask_name As String
   subtask_name = Trim(Text_子任务.Text)
   Dim subIdx As Integer = List_子任务.ListIndex
   
   Select Case nID
      Case 任务管理_YFbunGroup1_addsubtask ' 添加子任务
         If subtask_name = "" Then
            MsgBox("子任务名称不能为空！")
            Exit Function
         End If
         
         If taskType = 1 Then ' 周期任务
            With periodicTasks(taskIndex)
               Dim count As Integer = Val(.subtasks(0, 1)) ' 当前子任务数量
               ' 增加一行：行索引从 0 到 count+1，列固定为 0..1
               ReDim Preserve .subtasks(0 To count + 1, 0 To 1)
               ' 添加新子任务到新行
               .subtasks(count + 1, 0) = subtask_name
               .subtasks(count + 1, 1) = "0" ' 状态初始 0
               ' 更新元信息
               .subtasks(0, 0) = "1" ' 有子任务
               .subtasks(0, 1) = Str(count + 1)
            End With
         Else ' 普通任务
            With regularTasks(taskIndex)
               Dim count As Integer = Val(.subtasks(0, 1))
               ReDim Preserve .subtasks(0 To count + 1, 0 To 1)
               .subtasks(count + 1, 0) = subtask_name
               .subtasks(count + 1, 1) = "0"
               .subtasks(0, 0)         = "1"
               .subtasks(0, 1)         = Str(count + 1)
            End With
         End If
         
         ' 刷新子任务列表
         RefreshSubtaskList(taskType, taskIndex)
         ' 清空输入框
         Text_子任务.Text = ""
         
      Case 任务管理_YFbunGroup1_editsubtask ' 修改子任务
         If subIdx < 0 Then
            MsgBox("请先选中要修改的子任务！")
            Exit Function
         End If
         If subtask_name = "" Then
            MsgBox("子任务名称不能为空！")
            Exit Function
         End If
         
         Dim row As Integer = subIdx + 1 ' 子任务在数组中的行索引（行0为元信息）
         If taskType = 1 Then
            With periodicTasks(taskIndex)
               Dim count As Integer = Val(.subtasks(0, 1))
               If row > count Then Exit Function
               .subtasks(row, 0) = subtask_name
            End With
         Else
            With regularTasks(taskIndex)
               Dim count As Integer = Val(.subtasks(0, 1))
               If row > count Then Exit Function
               .subtasks(row, 0) = subtask_name
            End With
         End If
         
         ' 刷新列表并保持选中
         RefreshSubtaskList(taskType, taskIndex)
         If subIdx < List_子任务.ListCount Then List_子任务.ListIndex = subIdx
         Text_子任务.Text = ""
         
         
      Case 任务管理_YFbunGroup1_delsubtask ' 删除子任务
         If subIdx < 0 Then
            MsgBox("请先选中要删除的子任务！")
            Exit Function
         End If
         
         Dim row As Integer = subIdx + 1
         If taskType = 1 Then
            With periodicTasks(taskIndex)
               Dim count As Integer = Val(.subtasks(0, 1))
               If row > count Then Exit Function
               ' 将后续子任务向前移动
               For i As Integer = row To count - 1
                  .subtasks(i, 0) = .subtasks(i + 1, 0)
                  .subtasks(i, 1) = .subtasks(i + 1, 1)
               Next
               ' 减少一行
               ReDim Preserve .subtasks(0 To count - 1, 0 To 1)
               ' 更新元信息
               .subtasks(0, 1) = Str(count - 1)
               If count - 1 = 0 Then .subtasks(0, 0) = "0"   ' 无子任务
            End With
         Else
            With regularTasks(taskIndex)
               Dim count As Integer = Val(.subtasks(0, 1))
               If row > count Then Exit Function
               For i As Integer = row To count - 1
                  .subtasks(i, 0) = .subtasks(i + 1, 0)
                  .subtasks(i, 1) = .subtasks(i + 1, 1)
               Next
               ReDim Preserve .subtasks(0 To count - 1, 0 To 1)
               .subtasks(0, 1) = Str(count - 1)
               If count - 1 = 0 Then .subtasks(0, 0) = "0"
            End With
         End If
         
         RefreshSubtaskList(taskType, taskIndex)
         If List_子任务.ListCount > 0 Then List_子任务.ListIndex = 0
         
         
      Case 任务管理_YFbunGroup1_moveup ' 上移
         If subIdx <= 0 Then
            MsgBox("已经是第一个子任务，无法上移！")
            Exit Function
         End If
         
         Dim row As Integer = subIdx + 1
         Dim prevRow As Integer = row - 1
         If taskType = 1 Then
            With periodicTasks(taskIndex)
               Dim count As Integer = Val(.subtasks(0, 1))
               If row > count Then Exit Function
               ' 交换两行数据
               Swap .subtasks(row, 0), .subtasks(prevRow, 0)
               Swap .subtasks(row, 1), .subtasks(prevRow, 1)
            End With
         Else
            With regularTasks(taskIndex)
               Dim count As Integer = Val(.subtasks(0, 1))
               If row > count Then Exit Function
               Swap .subtasks(row, 0), .subtasks(prevRow, 0)
               Swap .subtasks(row, 1), .subtasks(prevRow, 1)
            End With
         End If
         
         RefreshSubtaskList(taskType, taskIndex)
         List_子任务.ListIndex = subIdx - 1 ' 选中上移后的项
         
         
      Case 任务管理_YFbunGroup1_movedown ' 下移
         Dim totalCount As Integer
         If taskType = 1 Then
            totalCount = Val(periodicTasks(taskIndex).subtasks(0, 1))
         Else
            totalCount = Val(regularTasks(taskIndex).subtasks(0, 1))
         End If
         If subIdx < 0 Or subIdx >= totalCount - 1 Then
            MsgBox("已经是最后一个子任务，无法下移！")
            Exit Function
         End If
         
         Dim row As Integer = subIdx + 1
         Dim nextRow As Integer = row + 1
         If taskType = 1 Then
            With periodicTasks(taskIndex)
               Swap .subtasks(row, 0), .subtasks(nextRow, 0)
               Swap .subtasks(row, 1), .subtasks(nextRow, 1)
            End With
         Else
            With regularTasks(taskIndex)
               Swap .subtasks(row, 0), .subtasks(nextRow, 0)
               Swap .subtasks(row, 1), .subtasks(nextRow, 1)
            End With
         End If
         
         RefreshSubtaskList(taskType, taskIndex)
         List_子任务.ListIndex = subIdx + 1
         
         
   End Select
   
   Function = TRUE '  如果你处理了事件，就返回 Return TRUE ,控件就不进行默认处理了。
   
End Function

'[任务管理.YFbunGroup2]事件 : 点击了某个按钮
'hWndForm  当前窗口的句柄(WIN系统用来识别窗口的一个编号，如果多开本窗口，必须 Me.hWndForm = hWndForm 后才可以执行后续操作本窗口的代码)
'nID       按钮号，鼠标点击了某个按钮，就是点击后释放鼠标时触发
Function 任务管理_YFbunGroup2_Command(hWndForm As hWnd, hWndControl As hWnd, nID As Long) As Long
   Dim idx       As Integer = List_全任务列表.ListIndex
   Dim itemData  As Long    = List_全任务列表.ItemData(idx)
   Dim taskType  As Integer = (itemData Shr 16) And &HFFFF
   Dim taskIndex As Integer = itemData          And &HFFFF
   
   Select Case nID
      Case 任务管理_YFbunGroup2_newtask ' 新建任务
         Dim newIndex As Integer
         
         ' 确定新任务的索引（当前数组长度）
         newIndex = UBound(periodicTasks) + 1
         ReDim Preserve periodicTasks(0 To newIndex)
         
         ' 初始化新任务
         With periodicTasks(newIndex)
            .id              = GenerateTaskId()
            .Name            = "新任务"
            .createTime      = Now()
            .PeriodType      = 1 ' 日
            .currentPeriod   = SetTimeTo4AM(DateTimePicker1.DateTime)
            .nextPeriod      = CalculateNextPeriod(.currentPeriod, .PeriodType)
            .completed       = 0
            .completionCount = 0
            .onList          = 0
            .tempStatus      = 0
            ReDim.subtasks(0 To 0, 0 To 1) ' 仅元信息行
            .subtasks(0, 0) = "0" ' 无子任务
            .subtasks(0, 1) = "0" ' 子任务数量为0
         End With
         
         ' 刷新任务列表
         RefreshFullTaskList()
         ' 自动选中新添加的任务
         ' 新任务的索引在列表中的位置：普通任务标题行(1)+普通任务数(UBound(regularTasks) + 1)+周期任务标题行(1) + 新任务索引(listIdx) + 1（因为列表索引从0开始）
         Dim listIdx As Integer
         listIdx = UBound(regularTasks) + newIndex + 3 ' 周期任务标题在第0行，第一个周期任务在第1行
         List_全任务列表.ListIndex = listIdx
         ' 触发选中事件，显示任务详情
         任务管理_List_全任务列表_LBN_SelChange(hWndForm, List_全任务列表.hWnd)
         
      Case 任务管理_YFbunGroup2_toShow ' 上榜
         
      Case 任务管理_YFbunGroup2_save ' 修改
         ' 验证是否选中了有效任务
         If idx < 0 Then
            MsgBox("请先选中一个任务！")
            Exit Function
         End If
         If itemData = -1 Then
            MsgBox("不能修改标题行！")
            Exit Function
         End If
         ModifyTask(taskType, taskIndex)
         RefreshFullTaskList()
      Case 任务管理_YFbunGroup2_archive ' 归档
         ' 验证是否选中了有效任务
         If idx < 0 Then
            MsgBox("请先选中一个任务！")
            Exit Function
         End If
         If itemData = -1 Then
            MsgBox("不能归档标题行！")
            Exit Function
         End If
         ArchiveTask(taskType, taskIndex)
         RefreshFullTaskList()
      Case 任务管理_YFbunGroup2_delete ' 删除
         ' 验证是否选中了有效任务
         If idx < 0 Then
            MsgBox("请先选中一个任务！")
            Exit Function
         End If
         If itemData = -1 Then
            MsgBox("不能删除标题行！")
            Exit Function
         End If
         
         ' 二次确认
         If MsgBox("确定要删除任务“" & IIf(taskType = 1, periodicTasks(taskIndex).Name, regularTasks(taskIndex).name) & "”吗？", MB_YESNO) <> IDYES Then
            Exit Function
         End If
         ' 执行删除
         If DeleteTask(taskType, taskIndex) Then
            ' 刷新全任务列表
            RefreshFullTaskList()
            ' 清空编辑界面
            Text_任务名.Text = ""
            Option1.Value    = False
            Option2.Value    = False
            Option3.Value    = False
            Option4.Value    = False
            Option5.Value    = False
            DateTimePicker1.DateTime = AfxVariantTimeToSystemTime(Now)
            List_子任务.Clear
            MsgBox("删除成功！")
         Else
            MsgBox("删除失败！")
         End If
   End Select
   Function = TRUE '  如果你处理了事件，就返回 Return TRUE ,控件就不进行默认处理了。
End Function
