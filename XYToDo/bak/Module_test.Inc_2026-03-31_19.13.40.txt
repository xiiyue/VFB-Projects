' ========== 测试加载功能 ==========
Sub TestLoad()
    Dim i As Long
    Dim j As Long

    Print "=== 测试加载功能 ==="
    
    ' 检查文件是否存在
    If Not FileExists(periodicFile) Or Not FileExists(deferredFile) Or Not FileExists(regularFile) Then
        Print "错误：文件不存在，请先运行 TestSave 创建数据文件。"
        Exit Sub
    End If
    
    Print "清空内存数组..."
    Erase periodicTasks
    Erase deferredTasks
    Erase regularTasks
    Print "数组已清空。"

    Print "正在加载数据..."
    LoadPeriodicTasks()
    LoadDeferredTasks()
    LoadRegularTasks()
    Print "加载完成。"

    ' ---------- 验证数据 ----------
    Print "验证数据..."

    ' 周期任务
    Dim periodCount As Long = UBound(periodicTasks) + 1
    Print "加载后周期任务数: " & periodCount
    For i = 0 To periodCount - 1
        With periodicTasks(i)
            Print "  任务" & i & ": " & .Name
            Print "    createTime: " & .createTime
            Print "    PeriodType: " & .PeriodType
            Print "    currentPeriod: " & .currentPeriod
            Print "    nextPeriod: " & .nextPeriod
            Print "    completed: " & .completed
            Print "    completionCount: " & .completionCount
            Print "    onList: " & .onList
            Print "    tempStatus: " & .tempStatus
            ' 子任务
            Dim subFlag As Integer = Val(.subtasks(0, 0))
            Dim subCount As Integer = Val(.subtasks(0, 1))
            Print "    子任务标记: " & subFlag & ", 数量: " & subCount
            If subFlag = 1 Then
                For j = 1 To subCount
                    Print "      子任务" & j & ": " & .subtasks(j, 0) & " -> " & .subtasks(j, 1)
                Next
            End If
        End With
    Next

    ' 延期任务
    Dim deferCount As Long = UBound(deferredTasks) + 1
    Print "加载后延期任务数: " & deferCount
    For i = 0 To deferCount - 1
        With deferredTasks(i)
            Print "  延期任务" & i & ": " & .name
            Print "    createTime: " & .createTime
            Print "    target: " & .target
            Print "    onList: " & .onList
            Print "    tempStatus: " & .tempStatus
            Dim subFlag As Integer = Val(.subtasks(0, 0))
            Dim subCount As Integer = Val(.subtasks(0, 1))
            Print "    子任务标记: " & subFlag & ", 数量: " & subCount
            If subFlag = 1 Then
                For j = 1 To subCount
                    Print "      子任务" & j & ": " & .subtasks(j, 0) & " -> " & .subtasks(j, 1)
                Next
            End If
        End With
    Next

    ' 普通任务
    Dim regularCount As Long = UBound(regularTasks) + 1
    Print "加载后普通任务数: " & regularCount
    For i = 0 To regularCount - 1
        With regularTasks(i)
            Print "  普通任务" & i & ": " & .name
            Print "    createTime: " & .createTime
            Print "    taskTime: " & .taskTime
            Print "    onList: " & .onList
            Print "    tempStatus: " & .tempStatus
            Dim subFlag As Integer = Val(.subtasks(0, 0))
            Dim subCount As Integer = Val(.subtasks(0, 1))
            Print "    子任务标记: " & subFlag & ", 数量: " & subCount
            If subFlag = 1 Then
                For j = 1 To subCount
                    Print "      子任务" & j & ": " & .subtasks(j, 0) & " -> " & .subtasks(j, 1)
                Next
            End If
        End With
    Next

    Print "测试加载结束。"
End Sub
